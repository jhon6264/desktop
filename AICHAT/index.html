Example Request

fetch("https://jcrpbot.gt.tc/api/bot.php", {
  method: "POST",
  headers: {
    "Content-Type": "application/json"
  },
  body: JSON.stringify({
    message: "who is jhon?",
    conversation_id: "optional_previous_id"
  })
})
  .then((response) => response.json())
  .then((data) => console.log(data));
